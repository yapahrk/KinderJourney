    function calculateAge(birthDate) {
      if (!birthDate) return 0;
      const birth = new Date(birthDate);
      const today = new Date();
      const months = (today.getFullYear() - birth.getFullYear()) * 12 + (today.getMonth() - birth.getMonth());
      return Math.max(0, months);
    }

    async function toggleChecklistItem(category, ageRange, itemId, isChecked, item) {
      const itemTitle = item.title || item.name || item.food || item.activity;
      const existingRecord = allRecords.find(r => 
        r.type === category && 
        r.category === ageRange && 
        r.item_title === itemTitle
      );

      const notesElement = document.getElementById(`notes-${itemId}`);
      const notes = notesElement ? notesElement.value : '';

      if (existingRecord) {
        existingRecord.completed = isChecked;
        existingRecord.completed_date = isChecked ? new Date().toISOString() : '';
        existingRecord.notes = notes;
        await handleUpdate(existingRecord);
      } else {
        const profileRecord = allRecords.find(r => r.type === 'profile');
        const newRecord = {
          id: `${category}-${Date.now()}-${Math.random()}`,
          type: category,
          child_name: profileRecord?.child_name || '',
          birth_date: profileRecord?.birth_date || '',
          age_months: calculateAge(profileRecord?.birth_date),
          category: ageRange,
          item_title: itemTitle,
          completed: isChecked,
          completed_date: isChecked ? new Date().toISOString() : '',
          notes: notes,
          created_at: new Date().toISOString()
        };
        await handleCreate(newRecord);
      }
    }
